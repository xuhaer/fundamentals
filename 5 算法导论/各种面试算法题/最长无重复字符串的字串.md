def longest_substring(str_: str) -> int:
    '''返回给定字符串对无重复字符的最长子串'''
    mem_dict = {} # 存储字符所在的下标
    max_len, start = 0, 0
    if not str_:
        return max_len
    temp_max = 0
    for i, s in enumerate(str_):
        if s in mem_dict and mem_dict[s] >= start:
            start = mem_dict[s] + 1
        temp_max = i - start + 1
        mem_dict[s] = i
        max_len = max(max_len, temp_max)
        print('mem_dict', mem_dict)
        print('max_len', max_len)
    return max_len

longest_substring('pwwkew') # 'wke': 3
